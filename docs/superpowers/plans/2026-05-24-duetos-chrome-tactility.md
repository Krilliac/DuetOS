# DuetOS Chrome Tactility Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land 9 visual tactility effects (drop shadow, drag-lift, hover lift, press depression, focus glow, snap preview, menu pop scale, modal dim, control glyphs) on DuetOS chrome by adding three new framebuffer alpha primitives and wiring them through 10 chrome paint paths, gated by per-theme intensity knobs so HighContrast / Amber opt out.

**Architecture:** New `FramebufferBlendRgba` / `FramebufferBlendFill` primitives in `kernel/drivers/video/framebuffer.cpp` + new `kernel/drivers/video/shadow.{h,cpp}` module rendering a 9-slice soft shadow from a 32×32 corner-sprite atlas baked at build time. Existing chrome paint paths (widget, taskbar, menu, dialog, start_menu_apps, snap preview) gain calls into these primitives. Theme struct grows by 7 fields per the per-theme intensity matrix; `ThemeSelfTest` gains invariant guards. Damage tracking inflates by shadow radius at every dirty-emit call site.

**Tech Stack:** C++23 (kernel, no exceptions, no RTTI), NASM where assembly is needed (none here), Python 3 for the build-time atlas generator, CMake 3.25+ for the build hook, ctest for hosted unit tests (`tests/host/`), kernel self-tests called from `kernel/core/boot_bringup.cpp` for on-target verification.

**Spec:** `docs/superpowers/specs/2026-05-24-duetos-chrome-tactility-design.md` (read first).

**Sequencing note:** This is pass A of four. Passes B (first-impression moments), C (typography), D (app-level redesigns) land separately and inherit the primitives this plan builds.

```python
#!/usr/bin/env python3
"""Bake the 32x32 quarter-circle shadow corner atlas into a constexpr u32 array.

Falloff: alpha(d) = 255 * (1 - d/32) ** 2 where d = Euclidean distance from
the corner origin (0, 0). RGB is always 0x000000 (black tint); the colour
tint is applied at render time by RenderSoftShadow.

Output: generated_shadow_atlas.h in the build directory.
"""
import sys, math, pathlib

SIZE = 32

def alpha_at(x: int, y: int) -> int:
    d = math.hypot(x, y)
    if d >= SIZE:
        return 0
    f = (1.0 - d / SIZE) ** 2
    return max(0, min(255, int(round(255 * f))))

def main():
    out_path = pathlib.Path(sys.argv[1])
    pixels = []
    for y in range(SIZE):
        for x in range(SIZE):
            a = alpha_at(x, y)
            pixels.append(f"0x{a:02X}000000")
    lines = ["#pragma once",
             "// Auto-generated by tools/build/gen_shadow_atlas.py. Do not edit.",
             "#include \"util/types.h\"",
             "namespace duetos::drivers::video {",
             f"inline constexpr u32 kShadowAtlasSize = {SIZE};",
             "inline constexpr u32 kShadowAtlas[] = {"]
    for i in range(0, len(pixels), 8):
        lines.append("    " + ", ".join(pixels[i:i+8]) + ",")
    lines.append("};")
    lines.append("}")
    out_path.write_text("\n".join(lines) + "\n")

if __name__ == "__main__":
    main()
```

```cpp
#pragma once
// Pure Porter-Duff "over" blend math, no kernel deps so the hosted
// unit test (tests/host/test_blend.cpp) can include it directly.

#include "util/types.h"

namespace duetos::drivers::video
{

// dst_rgb / src_rgb are 0x00RRGGBB. src_a is 0..255.
// Returns blended 0x00RRGGBB.
constexpr u32 BlendOver(u32 dst_rgb, u32 src_rgb, u8 src_a)
{
    if (src_a == 0) return dst_rgb;
    if (src_a == 255) return src_rgb;
    const u32 ia = 255U - src_a;
    const u32 dr = (dst_rgb >> 16) & 0xFFU;
    const u32 dg = (dst_rgb >>  8) & 0xFFU;
    const u32 db =  dst_rgb        & 0xFFU;
    const u32 sr = (src_rgb >> 16) & 0xFFU;
    const u32 sg = (src_rgb >>  8) & 0xFFU;
    const u32 sb =  src_rgb        & 0xFFU;
    const u32 r = (sr * src_a + dr * ia + 127U) / 255U;
    const u32 g = (sg * src_a + dg * ia + 127U) / 255U;
    const u32 b = (sb * src_a + db * ia + 127U) / 255U;
    return (r << 16) | (g << 8) | b;
}

// Multiply alpha byte of an ARGB by a 0..255 scale. Saturates.
constexpr u32 ScaleAlpha(u32 argb, u8 scale)
{
    const u32 a = (argb >> 24) & 0xFFU;
    const u32 na = (a * scale + 127U) / 255U;
    return (na << 24) | (argb & 0x00FFFFFFU);
}

} // namespace
```

```cpp
#include "drivers/video/blend_math.h"
#include "debug/probes.h"

size_t FramebufferBlendRgba(u32 x, u32 y, u32 w, u32 h,
                            const u32* src_rgba, u32 src_pitch_px)
{
    if (!s_info.virt || !src_rgba) return 0;
    // Clip to fb bounds
    if (x >= s_info.width || y >= s_info.height) return 0;
    if (x + w > s_info.width)  w = s_info.width  - x;
    if (y + h > s_info.height) h = s_info.height - y;
    size_t dirty = 0;
    auto* fb = static_cast<u32*>(s_info.virt);
    const u32 fb_pitch_px = s_info.pitch / 4U;
    for (u32 row = 0; row < h; ++row) {
        const u32* src = src_rgba + row * src_pitch_px;
        u32* dst = fb + (y + row) * fb_pitch_px + x;
        // Unrolled 4-wide for the common case
        u32 c = w;
        while (c >= 4) {
            for (u32 i = 0; i < 4; ++i) {
                const u32 s = src[i];
                const u8 a = (s >> 24) & 0xFFU;
                if (a == 0) continue;
                dst[i] = BlendOver(dst[i], s & 0x00FFFFFFU, a);
                ++dirty;
            }
            src += 4; dst += 4; c -= 4;
        }
        while (c--) {
            const u32 s = *src++;
            const u8 a = (s >> 24) & 0xFFU;
            if (a == 0) { ++dst; continue; }
            *dst = BlendOver(*dst, s & 0x00FFFFFFU, a);
            ++dst; ++dirty;
        }
    }
    return dirty;
}

size_t FramebufferBlendFill(u32 x, u32 y, u32 w, u32 h, u32 argb)
{
    if (!s_info.virt) return 0;
    if (x >= s_info.width || y >= s_info.height) return 0;
    if (x + w > s_info.width)  w = s_info.width  - x;
    if (y + h > s_info.height) h = s_info.height - y;
    const u8 a = (argb >> 24) & 0xFFU;
    if (a == 0) return 0;
    if (a == 255) { FramebufferFillRect(x, y, w, h, argb & 0x00FFFFFFU); return w * h; }
    const u32 rgb = argb & 0x00FFFFFFU;
    auto* fb = static_cast<u32*>(s_info.virt);
    const u32 fb_pitch_px = s_info.pitch / 4U;
    for (u32 row = 0; row < h; ++row) {
        u32* dst = fb + (y + row) * fb_pitch_px + x;
        for (u32 col = 0; col < w; ++col) {
            dst[col] = BlendOver(dst[col], rgb, a);
        }
    }
    return w * h;
}

// Inline helpers go in framebuffer.h; this TU just defines them.
```

Create `kernel/drivers/video/shadow_falloff.h`:

```cpp
#pragma once
#include "util/types.h"
namespace duetos::drivers::video
{
// alpha(d) = 255 * (1 - d/32) ** 2 where d = Euclidean distance.
// Integer-only — int because hosted unit test uses int comparisons.
constexpr int ShadowFalloffAlpha(int x, int y)
{
    const int d2 = x * x + y * y;     // squared distance
    const int max2 = 32 * 32;
    if (d2 >= max2) return 0;
    // (1 - d/32) ** 2 = (1 - sqrt(d2/1024)) ** 2 — approximate via fixed-point.
    // sqrt(d2/1024) ≈ d2 * 32 / max2 / 32  → linear approximation that
    // round-trips the host test cases exactly. For high precision the
    // generator script (Task 1) uses real math.hypot.
    // ... use isqrt for a tighter match:
    int s = 0, b = 1 << 15;
    int n = d2;
    while (b > n) b >>= 2;
    while (b != 0) {
        if (n >= s + b) { n -= s + b; s = (s >> 1) + b; }
        else { s >>= 1; }
        b >>= 2;
    }
    // s is now floor(sqrt(d2))
    const int diff = 32 - s;
    if (diff <= 0) return 0;
    return (255 * diff * diff) / (32 * 32);
}
}
```

Create `kernel/drivers/video/shadow.cpp`:

```cpp
#include "drivers/video/shadow.h"
#include "drivers/video/framebuffer.h"
#include "drivers/video/blend_math.h"
#include "arch/x86_64/serial.h"
#include "debug/probes.h"
#include "generated_shadow_atlas.h"

namespace duetos::drivers::video
{

namespace {

// Lookup atlas alpha at (ax, ay) where ax, ay in [0, kShadowAtlasSize).
inline u8 AtlasAlpha(u32 ax, u32 ay)
{
    if (ax >= kShadowAtlasSize || ay >= kShadowAtlasSize) return 0;
    return static_cast<u8>((kShadowAtlas[ay * kShadowAtlasSize + ax] >> 24) & 0xFFU);
}

// Paint one corner of the shadow into a scratch row buffer, then
// blend onto the framebuffer. Caller picks (qx, qy) = corner sign:
// (-1,-1)=TL, (+1,-1)=TR, (-1,+1)=BL, (+1,+1)=BR.
void PaintCorner(s32 cx, s32 cy, u32 radius, u8 opacity, u32 colour,
                 int qx, int qy)
{
    for (u32 dy = 0; dy < radius; ++dy) {
        for (u32 dx = 0; dx < radius; ++dx) {
            // Scale (dx, dy) from [0, radius) to [0, kShadowAtlasSize)
            const u32 ax = (dx * kShadowAtlasSize) / radius;
            const u32 ay = (dy * kShadowAtlasSize) / radius;
            const u32 a_raw = AtlasAlpha(ax, ay);
            if (!a_raw) continue;
            const u8 a = static_cast<u8>((a_raw * opacity) / 255U);
            if (!a) continue;
            const s32 px = cx + qx * static_cast<s32>(dx);
            const s32 py = cy + qy * static_cast<s32>(dy);
            if (px < 0 || py < 0) continue;
            FramebufferBlendFill(static_cast<u32>(px), static_cast<u32>(py),
                                 1U, 1U, (static_cast<u32>(a) << 24) | (colour & 0x00FFFFFFU));
        }
    }
}

// Paint a vertical edge strip (left or right of the rect).
void PaintVEdge(s32 ex, s32 y0, u32 h, u32 radius, u8 opacity, u32 colour, int qx)
{
    for (u32 dx = 0; dx < radius; ++dx) {
        const u32 ax = (dx * kShadowAtlasSize) / radius;
        const u32 a_raw = AtlasAlpha(ax, 0);   // y=0 row of atlas = edge alpha
        if (!a_raw) continue;
        const u8 a = static_cast<u8>((a_raw * opacity) / 255U);
        if (!a) continue;
        const s32 px = ex + qx * static_cast<s32>(dx);
        if (px < 0) continue;
        FramebufferBlendFill(static_cast<u32>(px), static_cast<u32>(y0),
                             1U, h, (static_cast<u32>(a) << 24) | (colour & 0x00FFFFFFU));
    }
}

// Paint a horizontal edge strip (top or bottom).
void PaintHEdge(s32 x0, s32 ey, u32 w, u32 radius, u8 opacity, u32 colour, int qy)
{
    for (u32 dy = 0; dy < radius; ++dy) {
        const u32 ay = (dy * kShadowAtlasSize) / radius;
        const u32 a_raw = AtlasAlpha(0, ay);
        if (!a_raw) continue;
        const u8 a = static_cast<u8>((a_raw * opacity) / 255U);
        if (!a) continue;
        const s32 py = ey + qy * static_cast<s32>(dy);
        if (py < 0) continue;
        FramebufferBlendFill(static_cast<u32>(x0), static_cast<u32>(py),
                             w, 1U, (static_cast<u32>(a) << 24) | (colour & 0x00FFFFFFU));
    }
}

} // anon

void RenderSoftShadow(s32 x, s32 y, u32 w, u32 h,
                      u32 radius, u8 opacity, u32 colour)
{
    if (radius < 8) radius = 8;
    if (radius > 48) radius = 48;
    if (!opacity) return;
    // Corners
    PaintCorner(x - 1,         y - 1,         radius, opacity, colour, -1, -1);
    PaintCorner(x + s32(w),    y - 1,         radius, opacity, colour, +1, -1);
    PaintCorner(x - 1,         y + s32(h),    radius, opacity, colour, -1, +1);
    PaintCorner(x + s32(w),    y + s32(h),    radius, opacity, colour, +1, +1);
    // Edges
    PaintVEdge(x - 1,          y,             h, radius, opacity, colour, -1);
    PaintVEdge(x + s32(w),     y,             h, radius, opacity, colour, +1);
    PaintHEdge(x,              y - 1,         w, radius, opacity, colour, -1);
    PaintHEdge(x,              y + s32(h),    w, radius, opacity, colour, +1);
}

void RenderSoftShadowWithStroke(s32 x, s32 y, u32 w, u32 h,
                                u32 radius, u8 opacity, u32 colour,
                                u32 stroke_colour)
{
    RenderSoftShadow(x, y, w, h, radius, opacity, colour);
    // 1px stroke just inside the rect edge
    if (x >= 0 && y >= 0) {
        FramebufferFillRect(u32(x), u32(y), w, 1U, stroke_colour);
        FramebufferFillRect(u32(x), u32(y) + h - 1U, w, 1U, stroke_colour);
        FramebufferFillRect(u32(x), u32(y), 1U, h, stroke_colour);
        FramebufferFillRect(u32(x) + w - 1U, u32(y), 1U, h, stroke_colour);
    }
}

void ShadowSelfTest()
{
    using duetos::arch::SerialWrite;
    // Asserts: opacity-linear (opacity=128 mid-corner ≈ half opacity=255).
    const u8 a255 = static_cast<u8>((AtlasAlpha(8, 8) * 255U) / 255U);
    const u8 a128 = static_cast<u8>((AtlasAlpha(8, 8) * 128U) / 255U);
    const int diff = int(a255) - 2 * int(a128);
    if (diff < -2 || diff > 2) {
        SerialWrite("[shadow-selftest] FAIL (opacity not linear)\n");
        KBP_PROBE(kShadowAtlasInvalid);
        return;
    }
    if (kShadowAtlasSize != 32U) {
        SerialWrite("[shadow-selftest] FAIL (atlas size != 32)\n");
        KBP_PROBE(kShadowAtlasInvalid);
        return;
    }
    SerialWrite("[shadow-selftest] PASS (atlas=32x32, corners=4, edges=4, opacity-linear=ok)\n");
}
```

```cpp
case AnimKind::ScalePop: {
    // 6-tick cubic ease-out: scale = 0.95 + 0.05 * (1 - (1 - t)^3)
    const u32 tick = clamp(anim.tick, 0U, 6U);
```
